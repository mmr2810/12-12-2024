#include <iostream>
#include <string.h>
using namespace std;

class AdditionalInfo {
    char address[20], license[20], insurance[20];
    long int contact;

public:
    AdditionalInfo() {
        strcpy(address, "XYZ");
        strcpy(license, "XY-0000000000");
        strcpy(insurance, "XY00000000X");
        contact = 0;
    }
    friend class Person;
};

class Person {
    char name[20], dob[10], blood[10];
    float height, weight;
    static int count;
    AdditionalInfo *info;

public:
    Person() {
        strcpy(name, "XYZ");
        strcpy(dob, "dd/mm/yy");
        strcpy(blood, "A+");
        height = weight = 0;
        info = new AdditionalInfo();
    }

    ~Person() { delete info; }

    void getData() {
        cout << "Enter Name: ";
        cin >> name;
        cout << "Enter DOB: ";
        cin >> dob;
        cout << "Enter Blood Group: ";
        cin >> blood;
        cout << "Enter Height: ";
        cin >> height;
        cout << "Enter Weight: ";
        cin >> weight;
        cout << "Enter Address: ";
        cin >> info->address;
        cout << "Enter License: ";
        cin >> info->license;
        cout << "Enter Insurance: ";
        cin >> info->insurance;
        cout << "Enter Contact: ";
        cin >> info->contact;
        count++;
    }

    void display() {
        cout << name << "\t" << dob << "\t" << blood << "\t" << height << "\t" << weight
             << "\t" << info->address << "\t" << info->license << "\t" << info->insurance << "\t" << info->contact << endl;
    }

    static void showCount() { cout << "Total Records: " << count << endl; }
};

int Person::count = 0;

int main() {
    int n;
    cout << "Enter number of records: ";
    cin >> n;

    Person *people = new Person[n];

    for (int i = 0; i < n; i++) {
        cout << "\nEnter details for person " << i + 1 << ":\n";
        people[i].getData();
    }

    cout << "\nDisplaying Records:\n";
    for (int i = 0; i < n; i++) {
        people[i].display();
    }

    Person::showCount();

    delete[] people;
    return 0;
}
